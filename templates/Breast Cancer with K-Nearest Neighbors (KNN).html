Tuning of KNN using Gridsearch¶

# Paramètre GridSearchCV 
k_neighbors_list = np.arange(1,30)

# Mes paramètres
search_param = {
    "n_neighbors": k_neighbors_list,
    "metric": ['euclidean', 'manhattan', 'minkowski', 'cityblock'],
    'weights': ['uniform', 'distance']
}

# Tentons d'utiser unn random Forest Classifier avec GridSearch pour trouver des paramètres optimales pour prédire
knn = KNeighborsClassifier()
grid = GridSearchCV(knn, search_param, cv = 12)
grid.fit(X_train, y_train)

# Affichons le score et les meilleurs paramètres 
print("Best Score: ", grid.best_score_)
print("Best Params: ", grid.best_params_)

model_knn =  grid.best_estimator_

In [65]:
# Prédict
#--------
y_pred_knn = model_knn.predict(X_test)

# Matrix de confusion
#--------------------
def confusion(y_true = y_test, y_pred = y_pred_knn, labels = [1,2]):
    conf_m = confusion_matrix(y_true = y_test, y_pred = y_pred_knn, labels = labels)  # [[TN, FP], [FN, TP]]
    print('-----------------')
    print(f'confusion matrix:\n{conf_m}')
    TN = conf_m[0,0]
    FP = conf_m[0,1]
    FN = conf_m[1,0]
    TP = conf_m[1,1]
    print('-----------------')
    print(f'sensitivity:{TP/(TP+FN)}')
    print('-----------------')
    print(f'specificity:{round(TN/(TN+FP),2)}') 
    
# Call the function
